public class aufgabe5 {
	boolean[][] schatzkarte;
	
	public int zaehleSchaetze() {
		int count = 0;
		
		for(int i = 0; i < schatzkarte.length; i++) {
			for(int j = 0; j < schatzkarte.length; j++) {
				if(schatzkarte[i][j]) {
					count++;
				}
			}
		}
		return count;
	    
	}
}

public class aufgabe2 {
	public boolean gueltigeEingabe(int zeile, int spalte) {
		int[][] offenesSpielfeld ;
		int n = offenesSpielfeld.length;
		
		return zeile >= 0 && zeile < n && spalte >= 0 && spalte < n && offenesSpielfeld[zeile][spalte] == 0;
	  
	}
}
